class Robot:

    # Constructor - the funtion that runs on the creation of an instance of the object
    def __init__(self, health, ad, ap, armor, mr):
        self.health = health
        self.ad = ad
        self.ap = ap
        self.armor = armor
        self.mr = mr


    def attack(self, target):
        target.damage(attack)

    def damage(self, dmg):
        mitigated = dmg - self.armor
        self.health -= mitigated

    def getHealth():
        return self.health




r1 = Robot(100, 10, 5)
r2 = Robot(120, 12, 0)

while r1.health > 0 and r2.health > 0:
    r1.attack(r2)
    r2.attack(r1)
    print(f"R1 health: {r1.getHealth()}")
    print(f"R2 health: {r2.getHealth()}")
